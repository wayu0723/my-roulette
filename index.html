    <h1>運試しルーレット</h1>

    <canvas id="canvas" width="300" height="300"></canvas><br>

    <button onclick="spin()">スタート！</button>


    <script>

        const canvas = document.getElementById('canvas');

        const ctx = canvas.getContext('2d');

        const items = ["あたり", "はずれ", "おやつ", "掃除当番", "もう一回"];

        const colors = ["#ffadad", "#ffd6a5", "#fdffb6", "#caffbf", "#9bf6ff"];

        let angle = 0;


        function draw() {

            const step = 360 / items.length;

            items.forEach((item, i) => {

                ctx.beginPath();

                ctx.fillStyle = colors[i % colors.length];

                ctx.moveTo(150, 150);

                ctx.arc(150, 150, 150, (step * i - 90) * Math.PI / 180, (step * (i + 1) - 90) * Math.PI / 180);

                ctx.fill();

                ctx.save();

                ctx.translate(150, 150);

                ctx.rotate((step * i + step / 2 - 90) * Math.PI / 180);

                ctx.fillStyle = "black";

                ctx.fillText(item, 70, 5);

                ctx.restore();

            });

        }


        function spin() {

            const addAngle = Math.floor(Math.random() * 360) + 1440; // 4回転以上

            angle += addAngle;

            canvas.style.transition = "transform 3s cubic-bezier(0.1, 0, 0.2, 1)";

            canvas.style.transform = `rotate(${angle}deg)`;

        }

        draw();

    </script>
